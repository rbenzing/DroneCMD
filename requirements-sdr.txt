# ===================================
# INSTALLATION INSTRUCTIONS
# ===================================
# 
# To install support for specific SDR hardware:
# 
# 1. RTL-SDR (included by default):
#    pip install -r requirements-minimal.txt
# 
# 2. HackRF only:
#    pip install -r requirements-minimal.txt pyhackrf
# 
# 3. Multiple SDR platforms via SoapySDR:
#    pip install -r requirements-minimal.txt SoapySDR
# 
# 4. All supported SDR platforms:
#    pip install -r requirements.txt
# 
# 5. Development with all SDR support:
#    pip install -r requirements-dev.txt
# 
# ===================================
# SYSTEM DEPENDENCIES
# ===================================
# 
# Some SDR drivers require system-level libraries:
# 
# Ubuntu/Debian:
#   sudo apt update
#   sudo apt install librtlsdr-dev libhackrf-dev libairspy-dev
#   sudo apt install libuhd-dev uhd-host  # For USRP
#   sudo apt install libiio-dev libad9361-dev  # For PlutoSDR
# 
# macOS (via Homebrew):
#   brew install librtlsdr hackrf airspy
#   brew install uhd  # For USRP
# 
# Windows:
#   Install Zadig for USB driver management
#   Download vendor-specific drivers and APIs
# 
# ===================================
# PERMISSION SETUP
# ===================================
# 
# Linux users may need udev rules for USB access:
# 
# RTL-SDR:
#   sudo wget -O /etc/udev/rules.d/20-rtlsdr.rules https://raw.githubusercontent.com/osmocom/rtl-sdr/master/rtl-sdr.rules
# 
# HackRF:
#   sudo wget -O /etc/udev/rules.d/53-hackrf.rules https://raw.githubusercontent.com/greatscottgadgets/hackrf/master/host/libhackrf/53-hackrf.rules
# 
# Then reload udev:
#   sudo udevadm control --reload-rules
#   sudo udevadm trigger
# 
# ===================================
# TESTING SDR INSTALLATION
# ===================================
# 
# After installation, test your SDR:
# 
# RTL-SDR:
#   rtl_test
#   python -c "from rtlsdr import RtlSdr; print('RTL-SDR OK')"
# 
# HackRF:
#   hackrf_info
#   python -c "import pyhackrf; print('HackRF OK')"
# 
# SoapySDR:
#   SoapySDRUtil --find
#   python -c "import SoapySDR; print('SoapySDR OK')"
